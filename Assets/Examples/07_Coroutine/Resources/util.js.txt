const CS = require("csharp");
function cs_generator(func, ...args) {
    const gen = func(args);
    let done = false;
    const next = function () {
        var move = gen.next();
        if (move.done)
            done = true;
        return move.value;
    };
    return CS.TestUtil.Generator(next, () => done);
}
exports.cs_generator = cs_generator;
